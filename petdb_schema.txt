--- USEFUL DERIVED RATIOS (compute inline in SQL) ---
  Nb/U  = nb / u      -- ~47 for MORB/OIB mantle; <20 suggests crustal contamination
  Ce/Pb = ce / pb     -- diagnostic mantle source ratio; high in OIB/MORB, low in arc/crust
  La/Sm = la / sm     -- MORB source depletion indicator
  Sm/Nd = sm / nd     -- parent/daughter ratio for Nd isotope systematics
  La/Yb = la / yb     -- REE fractionation and garnet signature
  Ba/Nb = ba / nb     -- High values indicate subduction fluid input
  Zr/Nb = zr / nb     -- Source depletion indicator
  Zr/Hf = zr / hf     -- ~36 in primitive mantle; deviations indicate fractionation
  Nb/Ta = nb / ta     -- ~17 in primitive mantle
  Rb/Sr = rb / sr     -- Fluid mobility ratio; elevated in arc settings
  Eu/Eu* = eu anomaly -- compute as eu / (sm * gd)^0.5 to detect plagioclase signature

EXAMPLE QUERIES:
  -- Complete REE pattern for a basin:
  SELECT basin_name, ROUND(AVG(la),2), ROUND(AVG(ce),2), ROUND(AVG(nd),2),
         ROUND(AVG(sm),2), ROUND(AVG(eu),2), ROUND(AVG(gd),2),
         ROUND(AVG(dy),2), ROUND(AVG(yb),2), ROUND(AVG(lu),2), COUNT(*) AS n
  FROM samples
  WHERE la IS NOT NULL AND yb IS NOT NULL
  GROUP BY basin_name ORDER BY n DESC;

  -- Ce/Pb ratio as mantle source discriminant:
  SELECT sample_name, basin_name, ce, pb, ROUND(ce/pb, 2) AS ce_pb
  FROM samples
  WHERE ce IS NOT NULL AND pb IS NOT NULL AND pb > 0
  ORDER BY ce_pb DESC;

  -- Primitive basalts using Ni, Cr, and MgO:
  SELECT sample_name, basin_name, mgo, ni, cr, sc
  FROM samples
  WHERE mgo > 10 AND ni IS NOT NULL AND cr IS NOT NULL
  ORDER BY mgo DESC;
